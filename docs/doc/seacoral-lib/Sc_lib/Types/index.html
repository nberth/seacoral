type strategy =
  1. | Nothing
    (*

    Do nothing.

    *)
  2. | Tool of string
    (*

    A single tool to start.

    *)
  3. | Parallel of strategy list
    (*

    Strategies to launch in parallel.

    *)
  4. | Sequence of strategy list
    (*

    Strategies to launch sequentially.

    *)

A strategy is a set of tool to start in a given order.
